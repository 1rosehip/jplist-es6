API: Refresh

Refresh API may be used to apply modified controls on content dynamically.

<script src="//cdnjs.cloudflare.com/ajax/libs/babel-polyfill/6.26.0/polyfill.min.js"></script>
<script src="jplist.min.js"></script>
<script>
    (function(){
        'use strict';

        //start jPList
        jplist.init();

        // DO SOME LOGIC...

        //refresh jPList
        jplist.refresh();
    })();
</script>
</html>

It also can be applied to the specific content group:

<script src="//cdnjs.cloudflare.com/ajax/libs/babel-polyfill/6.26.0/polyfill.min.js"></script>
<script src="jplist.min.js"></script>
<script>
    (function(){
        'use strict';

        //start jPList
        jplist.init();

        // DO SOME LOGIC...

        //refresh jPList
        jplist.refresh(groupName);
    })();
</script>
</html>

Also the element that triggered the event can be provided:

<script src="//cdnjs.cloudflare.com/ajax/libs/babel-polyfill/6.26.0/polyfill.min.js"></script>
<script src="jplist.min.js"></script>
<script>
    (function(){
        'use strict';

        //start jPList
        jplist.init();

        // DO SOME LOGIC...

        //refresh jPList
        jplist.refresh(groupName, targetElement);
    })();
</script>
</html>
